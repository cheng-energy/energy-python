```python
#CRAPS赌博游戏
'''CRAPS又称花旗骰，是美国拉斯维加斯非常受欢迎的一种的桌上赌博游戏。该游戏使用两粒骰子，玩家通过摇两粒骰子获得点数进行游戏。简化后的规则是：玩家第一次摇骰子如果摇出了 7 点或 11 点，玩家胜；玩家第一次如果摇出 2 点、3 点或 12 点，庄家胜；玩家如果摇出其他点数则游戏继续，玩家重新摇骰子，如果玩家摇出了 7 点，庄家胜；如果玩家摇出了第一次摇的点数，玩家胜；其他点数玩家继续摇骰子，直到分出胜负。为了增加代码的趣味性，我们设定游戏开始时玩家有 1000 元的赌注，每局游戏开始之前，玩家先下注，如果玩家获胜就可以获得对应下注金额的奖励，如果庄家获胜，玩家就会输掉自己下注的金额。游戏结束的条件是玩家破产（输光所有的赌注）。'''
import random
money = 1000
while money > 0:
    while True:
        print(f'你的资产目前有{money}元')
        print('每次下注不得超过自己的资产，不可为0')
        m = int(input('请下注：'))
        if 0 < m <= money:
            break
    first_point=random.randint(1,7)+random.randint(1,7)
    print(type(first_point))
    if first_point == 7 or first_point ==11:
        money+=m
        print('游戏结束，玩家获胜')
        print(f'你的资产现在是{money}')
    elif first_point == 2 or first_point ==12 or first_point ==3:
        money-=m
        print('游戏结束，庄家获胜')
        print(f'你的资产现在是{money}')
    else:
        print('游戏继续')
        while True:
            _point = random.randint(1,7)+random.randint(1,7)
            if _point == 7:
                money -= m
                print('游戏结束，庄家获胜')
                print(f'你的资产是{money}')
                break
            elif _point == first_point:
                print('游戏结束，玩家获胜')
                money+=m
                print(f'你的资产是{money}元')
                break
print('游戏结束，你破产了')
```

```python
#将一颗色子掷6000次，统计每种点数出现的次数
import random
point = [0]*6
for _ in range(6000):#下面的代码循环6000次
    face = random.randint(1,6)#例如face=2那么列表中2-1位置处的元素就＋1
    point[face-1] += 1
for face in range(1,7):
    print(f"点数'{face}'出现的次数是'{point[face-1]}'")
```
## 注意，random.randint(a,b)取值范围是左闭右闭区间
